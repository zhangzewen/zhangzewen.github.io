client-go 源码分析之 controller

Created: Aug 29, 2024 · 2 min read

本篇文章基于 client-go release-1.25 分支

Controller是一个接口，该接口的实例的主要作用是构建和启动 reflector。


type Controller interface {
  Run(stopCh <-chan struct{})
  HasSynced() bool
  LastSyncResourceVersion() string
}

type controller struct {
  config         Config
  reflector      *Reflector
  reflectorMutex sync.RWMutex
  clock          clock.Clock
}

controller 实现了上述的 Controller，Config 存储的是配置信息， 通过调用 New 构造 controller 实例：

// New makes a new Controller from the given Config.
func New(c *Config) Controller {
  // 只是简单地传递了配置，后面用于构建reflector实例
  ctlr := &controller{
    config: *c,
    clock:  &clock.RealClock{},
  }
  return ctlr
}

在实例化接口 Controller 后，通过 Run 来启动实例， 这里可以看到传入的 Config 配置信息会构造 reflector。

func (c *controller) Run(stopCh <-chan struct{}) {
  defer utilruntime.HandleCrash()
  go func() {
    <-stopCh
    c.config.Queue.Close()
  }()
  // 构建 Reflector
  r := NewReflector(
    c.config.ListerWatcher,
    c.config.ObjectType,
    c.config.Queue,
    c.config.FullResyncPeriod,
  )
  r.ShouldResync = c.config.ShouldResync
  r.WatchListPageSize = c.config.WatchListPageSize
  r.clock = c.clock
  if c.config.WatchErrorHandler != nil {
    r.watchErrorHandler = c.config.WatchErrorHandler
  }

  c.reflectorMutex.Lock()
  c.reflector = r
  c.reflectorMutex.Unlock()

  var wg wait.Group
  // 调用Reflector.Run方法，启动Reflector组件
  wg.StartWithChannel(stopCh, r.Run)
  // 调用controller.processLoop从controller.config.Queue中pop出item进行处理
  wait.Until(c.processLoop, time.Second, stopCh)
  wg.Wait()
}

从上面的代码中核心逻辑是 processLoop：

func (c *controller) processLoop() {
  for {
    obj, err := c.config.Queue.Pop(PopProcessFunc(c.config.Process))
    if err != nil {
      if err == ErrFIFOClosed {
        return
      }
      if c.config.RetryOnError {
        // This is the safe way to re-enqueue.
        c.config.Queue.AddIfNotPresent(obj)
      }
    }
  }
}

上述代码逻辑很明了： 从队列中 Pop 出数据进行处理，处理的逻辑是调用 controller.config.Process，这个最终还是会调用 processDeltas，其流程是遍历一个资源对象的修改事件列表，判断该对象在 Store 实例中是否存在, 不存在则创建, 反之则更新。 然后根据事件的类型, 调用对应的用户注册的回调函数。

//k8s.io/client-go/tools/cache/shared_informer.go
func (s *sharedIndexInformer) HandleDeltas(obj interface{}) error {
	s.blockDeltas.Lock()
	defer s.blockDeltas.Unlock()

	if deltas, ok := obj.(Deltas); ok {
		return processDeltas(s, s.indexer, deltas)
	}
	return errors.New("object given as Process argument is not Deltas")
}


// k8s.io/client-go/tools/cache/controller.go
func processDeltas(
	handler ResourceEventHandler,
	clientState Store,
	deltas Deltas,
) error {
	for _, d := range deltas {
		obj := d.Object

		switch d.Type {
		case Sync, Replaced, Added, Updated:
            // 存在，则更新
			if old, exists, err := clientState.Get(obj); err == nil && exists {
				if err := clientState.Update(obj); err != nil {
					return err
				}
                // 调用用户注册回调
				handler.OnUpdate(old, obj)
			} else {
                // 不存在，创建
				if err := clientState.Add(obj); err != nil {
					return err
				}
                // 调用用户注册回调
				handler.OnAdd(obj)
			}
		case Deleted:
            // 删除
			if err := clientState.Delete(obj); err != nil {
				return err
			}
            // 调用用户注册回调
			handler.OnDelete(obj)
		}
	}
	return nil
}
