kubernetes csi 之 node-driver-register

Created: Aug 2, 2025 · 2 min read

node-driver-register 是一个辅助容器，通过 CSI 接口 提供的 Identity Service 来获取插件信息，并且通过kubelet 插件注册接口 注册到 kubelet。所以，node-dirver-rigister 即是一个客户端，也是一个服务端。

和 kubelet 通信所使用的 unix domain socket：

  • 由 node-driver-register 创建的。
  • 通过 hostpath 类型的 volume 暴露在 kubernetes 的节点上，路径一般是： 
    /var/lib/kubelet/plugins_registry/<drivername.example.com>-reg.sock`
  • 且挂载到 Pod 中的路径必须是 /registration。

和第三方 CSI 插件通信的 unix domain socket：

  • 由 CSI Driver 创建。
  • 通过 hostpath 的 volume 暴露在 kubernetes 的节点上，路径一般是： 
    /var/lib/kubelet/plugins/<drivername.example.com>/csi.sock
  • 且挂载到 Pod 中的路径必须是 /csi。

整个 main 函数比较简单：

  • 获取 第三方 CSI 插件信息。
  • 把 第三方 CSI 插件注册到 kubelet 中。
func main() {
    // ...
    // 构建与第三方 CSI 插件服务的链接
	csiConn, err := connection.ConnectWithoutMetrics(ctx, *csiAddress)
	if err != nil {
		logger.Error(err, "Error connecting to CSI driver")
		klog.FlushAndExit(klog.ExitFlushTimeout, 1)
	}

	logger.V(1).Info("Calling CSI driver to discover driver name")
	getNameCtx, cancel := context.WithTimeout(ctx, *operationTimeout)
	defer cancel()

    // 通过 CSI 接口提供的 Identity Service 来获取插件信息
	csiDriverName, err := csirpc.GetDriverName(getNameCtx, csiConn)
	if err != nil {
		logger.Error(err, "Error retreiving CSI driver name")
		klog.FlushAndExit(klog.ExitFlushTimeout, 1)
	}
	logger.V(2).Info("CSI driver name", "csiDriverName", csiDriverName)
	defer closeGrpcConnection(logger, *csiAddress, csiConn)

	// Run forever
    // 向 kubelet 注册 csi 插件
	nodeRegister(ctx, csiDriverName, addr)
}

nodeRegister 通过 kubelet 插件注册接口 来生成 gPRC 服务，kubelet 通过监听 unix domain socket 文件的创建/更新 来生成 gRPC 客户端并调用 GetInfo 和 NotifyRegistrationStatus 把 CSI 插件注册到 kubelet 中。

func nodeRegister(ctx context.Context, csiDriverName, httpEndpoint string) {
	logger := klog.FromContext(ctx)
    // 创建插件注册服务，供kubelet 请求
	registrar := newRegistrationServer(csiDriverName, *kubeletRegistrationPath, supportedVersions)
    // 生成提供服务的 unix domain socket 的文件路径
	socketPath := buildSocketPath(csiDriverName)
    // 删除旧的 unix domain socket 文件，例如重启情况的发生
	if err := util.CleanupSocketFile(socketPath); err != nil {
		logger.Error(err, "")
		klog.FlushAndExit(klog.ExitFlushTimeout, 1)
	}

	var oldmask int
	if runtime.GOOS == "linux" {
		// Default to only user accessible socket, caller can open up later if desired
		oldmask, _ = util.Umask(0077)
	}

	logger.Info("Starting Registration Server", "socketPath", socketPath)
	lis, err := net.Listen("unix", socketPath)
	if err != nil {
		logger.Error(err, "Failed to listen on socket", "socketPath", socketPath)
		klog.FlushAndExit(klog.ExitFlushTimeout, 1)
	}
	if runtime.GOOS == "linux" {
		util.Umask(oldmask)
	}

	logger.Info("Registration Server started", "socketPath", socketPath)
	opts := []grpc.ServerOption{
		grpc.UnaryInterceptor(func(ctx context.Context, req interface{}, _ *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (resp interface{}, err error) {
			return handler(klog.NewContext(ctx, logger), req)
		}),
	}
    // 创建gRPC 服务
	grpcServer := grpc.NewServer(opts...)

	registerapi.RegisterRegistrationServer(grpcServer, registrar)
    //  httpServer 给 node-driver-register 添加了健康检查，以及 profile 的http服务，利于debug。
	go httpServer(ctx, socketPath, httpEndpoint, csiDriverName)
    // 获取到 kill 信号时，删除unix domain socket 文件,服务退出
	go removeRegSocket(logger, csiDriverName)
	if err := grpcServer.Serve(lis); err != nil {
		logger.Error(err, "Registration Server stopped serving")
		klog.FlushAndExit(klog.ExitFlushTimeout, 1)
	}

	os.Exit(0)
}
